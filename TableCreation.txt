CREATE TABLE orders (OrderID INT(4) UNSIGNED AUTO_INCREMENT PRIMARY KEY,
sandwich VARCHAR(15) NOT NULL,
snack VARCHAR (10) NOT NULL,
fruit VARCHAR(7) NOT NULL,
drink VARCHAR(8) NOT NULL,
Date date NOT NULL);

CREATE TABLE pupilorders (OrderID INT(4) UNSIGNED AUTO_INCREMENT PRIMARY KEY,
UserID INT(4) UNSIGNED AUTO_INCREMENT PRIMARY KEY,
completed tinyint(1));

CREATE TABLE pupils (UserID INT(4) UNSIGNED AUTO_INCREMENT PRIMARY KEY,
forename VARCHAR(20) NOT NULL,
surname VARCHAR(20) NOT NULL,
house VARCHAR(20) NOT NULL,
email VARCHAR(40) NOT NULL,
password VARCHAR(20) NOT NULL);

CREATE TABLE stock (FoodID INT(3) UNSIGNED AUTO_INCREMENT PRIMARY KEY,
FoodName VARCHAR(40) NOT NULL,
Quantity INT(4) NOT NULL,
category VARCHAR(40) NOT NULL);
